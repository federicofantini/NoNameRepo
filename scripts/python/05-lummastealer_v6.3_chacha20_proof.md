from Crypto.Cipher import ChaCha20

key = b"\x9E\xE9\x81\xA3\xCB\xFD\x24\x71\x69\xD3\xEE\x3F\x44\x58\x86\x2A\x37\x40\xBB\x06\x63\xB9\x55\xE2\x6F\x9C\x9D\x67\x9C\x48\xB0\x1F"
nonce = b"\x16\xD9\x70\x49\xFE\x34\xA8\x08"
ciphertext = b"\x09\x52\x89\x2D\x39\xA2\x3A\x87\x07\x04\xE7\x12\x7C\x43\x7F\x8E\x85\xE2\x61\x03\x6E\xAB\xBD\xF4\x00\x4C\xAC\x24\x66\x6F\x59\x5C\x40\x59\x28\xCA\x6B\x64\x36\x02\xE2\x79\xB5\xB2\xCF\xA2\x1B\x00\xC7\x2F\x51\x84\x5A\x8D\xB5\xCD\x0E\x72\x18\xC6\x9A\x45\x18\xB1\x60\xCD\x4E\xF8\x63\x1B\xCE\xD0\x7D\x96\x78\x2B\x29\x28\x49\x85\x86\x9B\x4B\xB1\xB8\x59\x85\xC9\x8B\x22\x79\x58\xB3\x92\xBA\x83\x16\xE7\x99\x5A\x97\x43\x85\x33\xA1\xAA\x3B\xE6\x29\x9D\xE7\x34\x17\xDD\xD7\xAA\xF1\x46\x75\x25\xBF\xF4\xAC\xAB\x78\x2A\x65\x03"

decipher = ChaCha20.new(key=key, nonce=nonce)
for c in range(1,10):
    print(f"counter={c}: {decipher.decrypt(ciphertext)}")
